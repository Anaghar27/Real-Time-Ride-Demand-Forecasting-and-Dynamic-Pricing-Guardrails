# 3) Timing helper
run_timed () {
  label="$1"; shift
  start=$(date +%s)
  echo "=== $label : START $(date) ==="
  "$@"
  rc=$?
  end=$(date +%s)
  echo "=== $label : END $(date) | elapsed_sec=$((end-start)) | rc=$rc ==="
  return $rc
}
